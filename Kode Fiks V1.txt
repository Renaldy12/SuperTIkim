/**
 * TRIGGER UTAMA: On Edit
 */
function mainTrigger(e) {
  if (!e) return;
  const ss = e.source;
  const sheet = ss.getActiveSheet();
  const range = e.range;
  const row = range.getRow();
  const col = range.getColumn();
  const val = range.getValue();
  const sheetName = sheet.getName();
  
  // Main Trigger Otomatis - Logika Urut Otomatis Konten pada Sheet Kalender
  if (sheetName === "Kalender") {
    if (col === 10 || col === 11) {
      urutkanDinamisKolomKonten();
      if (typeof updatePriorityList === "function") updatePriorityList();
    }
    // BLOK UNTUK KOLOM REQUEST (L & M)
    else if (col === 12 || col === 13) {
      urutkanDinamisKomponenRequest(); // Memanggil fungsi versi kebal yang baru
      if (typeof updatePriorityList === "function") updatePriorityList();
    }
    return;
  }



  // Main Trigger Otomatis - Logika jika bulan pada kalender berubah => set update konten prioritas pada dashboard
  if (sheetName === CONFIG.SHEET_KALENDER && (col === 11 || col === 13)) {
    updatePriorityList();
    return;
  }

  // Main Trigger Otomatis - Semua Logika pada Sheet Brief
  if (sheetName === CONFIG.SHEET_BRIEF && row >= 4) {
    const oldVal = e.oldValue ? e.oldValue.toString().trim() : ""; 
    const newVal = val ? val.toString().trim() : ""; 
    const dbSheet = ss.getSheetByName(CONFIG.SHEET_DATABASE);
    const kalenderSheet = ss.getSheetByName(CONFIG.SHEET_KALENDER);
    const judulKonten = sheet.getRange(row, CONFIG.KOLOM.JUDUL).getValue();


    // 1. RESET JIKA JUDUL DIHAPUS
    if (col === CONFIG.KOLOM.JUDUL && (newVal === "" || newVal === null)) {
      sheet.getRange(row, CONFIG.KOLOM.TIPE).clearContent(); 
      sheet.getRange(row, 3, 1, 7).clearContent();           
      if (oldVal) resetFormatKalender(kalenderSheet, oldVal);
      updatePriorityList();
      return;
    }

    // 2. INPUT JUDUL / TENGGAT (SINKRON & URUTKAN)
    if ((col === CONFIG.KOLOM.JUDUL || col === CONFIG.KOLOM.TENGGAT) && newVal !== "") {
      if (col === CONFIG.KOLOM.JUDUL) sinkronisasiKeKalender(sheet, kalenderSheet, row, newVal);
      SpreadsheetApp.flush();
      urutkanBrief(sheet);
      updatePriorityList();
    }

    if (!judulKonten) return;

    // 3. LOGIKA REVISI (Kirim ke PIC)
    if (col === CONFIG.KOLOM.TINDAK_LANJUT && newVal === "Revisi" && newVal !== oldVal) {
      if (sheet.getRange(row, CONFIG.KOLOM.PENGERJAAN).getValue() === "Selesai") {
        sheet.getRange(row, CONFIG.KOLOM.PENGERJAAN).setValue("On Progress");
        const namaPIC = sheet.getRange(row, CONFIG.KOLOM.PIC).getValue();
        kirimKePIC(namaPIC, `⚠️ <b>REVISI GES</b>\nHalo <b>${namaPIC}</b>, konten <b>${judulKonten}</b> perlu diperbaiki.`, dbSheet);
      }
    }

    // 4. LOGIKA SELESAI (Reset Revisi & Lapor Kepala)
    if (col === CONFIG.KOLOM.PENGERJAAN && newVal === "Selesai" && newVal !== oldVal) {
      sheet.getRange(row, CONFIG.KOLOM.TINDAK_LANJUT).clearContent(); // Reset kolom Revisi
      const namaPIC = sheet.getRange(row, CONFIG.KOLOM.PIC).getValue();
      kirimKeJabatan(CONFIG.JABATAN_KEPALA, `🔔 <b>KONTEN SELESAI</b>\nIzin melaporkan, konten <b>${judulKonten}</b> dari <b>${namaPIC}</b> sudah selesai, mohon tinjau ulang.`, dbSheet);
    }

    // 5. LOGIKA ACC (Format Kalender & Broadcast Staf)
    if (col === CONFIG.KOLOM.TINDAK_LANJUT && newVal === "ACC" && newVal !== oldVal) {
      setFormatKalenderACC(kalenderSheet, judulKonten);
      updatePriorityList(); 
      broadcastSemua(`🚀 <b>GAS UPLOAD</b>\nKonten <b>${judulKonten}</b> sudah disetujui!`, dbSheet);
    }
  }
}

/**
 * FUNGSI: Mengurutkan Kolom J & K.
 * AMAN: Tidak akan menghapus ketikan yang baru diisi setengah (menunggu tanggal/judul lengkap).
 */
function urutkanDinamisKolomKonten() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const activeRow = sheet.getActiveCell().getRow();
  
  SpreadsheetApp.flush();
  Utilities.sleep(100); 

  const lastRow = sheet.getLastRow();
  if (lastRow < 4) return;
  
  const dataJ = sheet.getRange(1, 10, lastRow, 1).getValues(); 
  
  let startRow = 0;
  let endRow = 0;
  
  for (let i = activeRow - 1; i >= 0; i--) {
    let cellVal = dataJ[i][0] ? dataJ[i][0].toString().trim() : "";
    if (cellVal === "Tanggal Posting") {
      startRow = i + 2; 
      break;
    }
  }
  
  if (startRow === 0) return;
  
  for (let i = startRow; i < lastRow; i++) {
    let cellVal = dataJ[i][0] ? dataJ[i][0].toString().trim() : "";
    if (cellVal.includes("Ide Konten Bulan") || (cellVal === "Tanggal Posting" && i > activeRow)) {
      endRow = i - 1;
      break;
    }
  }
  
  if (endRow === 0) endRow = lastRow + 10;
  if (activeRow > endRow) endRow = activeRow;

  if (startRow > 0 && endRow >= startRow) {
    const numRows = endRow - startRow + 1;
    const rangeBlok = sheet.getRange(startRow, 10, numRows, 2);
    
    const dataDisplay = rangeBlok.getDisplayValues(); 
    const dataFormulas = sheet.getRange(startRow, 10, numRows, 1).getFormulas(); 
    const dataKonten = sheet.getRange(startRow, 11, numRows, 1).getValues(); 
    
    let listTerisi = [];
    let listSetengahMatang = []; // Penampung data yang baru diisi sebelah
    
    for (let i = 0; i < numRows; i++) {
      let tglTeks = dataDisplay[i][0] ? dataDisplay[i][0].toString().trim() : "";
      let rumusJ = dataFormulas[i][0] ? dataFormulas[i][0] : "";
      let kontenK = dataKonten[i][0] ? dataKonten[i][0].toString().trim() : "";
      
      if (rumusJ === "" && sheet.getRange(startRow + i, 10).getValue() !== "") {
        rumusJ = sheet.getRange(startRow + i, 10).getValue();
      }
      
      // KUNCI PENGAMAN: Jika dua-duanya lengkap, masukkan ke daftar sortir
      if (kontenK !== "" && tglTeks !== "") {
        let parts = tglTeks.split("/");
        let urutanBobot = 0;
        
        if (parts.length === 3) {
          let d = new Date(2026, Number(parts[1]) - 1, Number(parts[0]));
          urutanBobot = d.getTime();
        } else {
          let d = new Date(tglTeks);
          if (!isNaN(d.getTime())) urutanBobot = new Date(2026, d.getMonth(), d.getDate()).getTime();
        }
        
        if (urutanBobot > 0) {
          listTerisi.push({ bobot: urutanBobot, rumus: rumusJ, konten: kontenK });
        }
      } 
      // JIKA BARU DIISI SEBELAH (Jangan dibuang, selamatkan ketikan user!)
      else if (kontenK !== "" || tglTeks !== "") {
        listSetengahMatang.push({ rumus: rumusJ, konten: kontenK });
      }
    }
    
    // Bersihkan tabel bulan sebelum ditulis ulang
    sheet.getRange(startRow, 10, numRows, 2).clearContent();
    
    // Urutkan data yang sudah berpasangan lengkap
    listTerisi.sort((a, b) => a.bobot - b.bobot);
    
    let currentIndex = 0;
    
    // Tulis dulu data yang sudah urut rapi di bagian atas
    listTerisi.forEach((item) => {
      let targetRow = startRow + currentIndex;
      if (item.rumus.toString().startsWith("=")) {
        sheet.getRange(targetRow, 10).setFormula(item.rumus);
      } else {
        sheet.getRange(targetRow, 10).setValue(item.rumus);
      }
      sheet.getRange(targetRow, 11).setValue(item.konten);
      currentIndex++;
    });
    
    // Tulis data setengah matang tepat di bawahnya agar tidak hilang
    listSetengahMatang.forEach((item) => {
      let targetRow = startRow + currentIndex;
      if (item.rumus) {
        if (item.rumus.toString().startsWith("=")) sheet.getRange(targetRow, 10).setFormula(item.rumus);
        else sheet.getRange(targetRow, 10).setValue(item.rumus);
      }
      if (item.konten) sheet.getRange(targetRow, 11).setValue(item.konten);
      currentIndex++;
    });
  }
  //SpreadsheetApp.getActiveSpreadsheet().toast("📋 Konten Update✅");
}

/**
 * FUNGSI: Mengurutkan Kolom L & M.
 * AMAN: Ketikan manual yang baru diisi setengah tidak akan dihapus oleh sistem.
 */
function urutkanDinamisKomponenRequest() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const activeRow = sheet.getActiveCell().getRow();
  
  const lastRow = sheet.getLastRow();
  if (lastRow < 4) return;
  
  const dataJ = sheet.getRange(1, 10, lastRow, 1).getValues(); 
  
  let startRow = 0;
  for (let i = activeRow - 1; i >= 0; i--) {
    let cellVal = dataJ[i][0] ? dataJ[i][0].toString().trim() : "";
    if (cellVal === "Tanggal Posting") {
      startRow = i + 2; 
      break;
    }
  }
  
  if (startRow === 0) return;
  
  let endRow = activeRow + 5;
  for (let i = startRow; i < lastRow; i++) {
    let cellVal = dataJ[i][0] ? dataJ[i][0].toString().trim() : "";
    if (cellVal.includes("Ide Konten Bulan") || (cellVal === "Tanggal Posting" && i > activeRow)) {
      endRow = i - 1;
      break;
    }
  }
  
  if (endRow < activeRow) endRow = activeRow;

  const numRows = endRow - startRow + 1;
  const rangeBlok = sheet.getRange(startRow, 12, numRows, 2);
  const dataValues = rangeBlok.getValues();
  const dataDisplay = rangeBlok.getDisplayValues(); 
  
  let listTerisi = [];
  let listSetengahMatang = []; // Penampung ketikan baru yang belum lengkap
  
  for (let i = 0; i < numRows; i++) {
    let rawTgl = dataValues[i][0];
    let tglTeks = dataDisplay[i][0] ? dataDisplay[i][0].toString().trim() : "";
    let requestValue = dataValues[i][1] ? dataValues[i][1].toString().trim() : "";
    
    // JIKA LENGKAP
    if (tglTeks !== "" && requestValue !== "") {
      let urutanBobot = 0;
      let tglFix = tglTeks;
      
      if (rawTgl instanceof Date && !isNaN(rawTgl.getTime())) {
        urutanBobot = new Date(2026, rawTgl.getMonth(), rawTgl.getDate()).getTime();
        tglFix = Utilities.formatDate(rawTgl, Session.getScriptTimeZone(), "dd/MM/yyyy");
      } else {
        let pemisah = tglTeks.includes("/") ? "/" : (tglTeks.includes("-") ? "-" : ".");
        let parts = tglTeks.split(pemisah);
        
        if (parts.length === 3) {
          let hari = Number(parts[0]);
          let bulan = Number(parts[1]);
          if (parts[0].length === 4) { hari = Number(parts[2]); bulan = Number(parts[1]); }
          urutanBobot = new Date(2026, bulan - 1, hari).getTime();
        } else {
          let d = new Date(tglTeks);
          if (!isNaN(d.getTime())) urutanBobot = new Date(2026, d.getMonth(), d.getDate()).getTime();
        }
      }
      
      if (urutanBobot > 0) {
        listTerisi.push({ bobot: urutanBobot, tglAsli: tglFix, request: requestValue });
      }
    } 
    // JIKA BARU DIKETIK SALAH SATU (Amankan data ketikan teks atau tanggalnya)
    else if (tglTeks !== "" || requestValue !== "") {
      listSetengahMatang.push({ tglAsli: tglTeks, request: requestValue });
    }
  }
  
  sheet.getRange(startRow, 12, numRows, 2).clearContent();
  
  listTerisi.sort((a, b) => a.bobot - b.bobot);
  
  let currentIndex = 0;
  
  // Tulis data lengkap yang sudah urut
  listTerisi.forEach((item) => {
    let targetRow = startRow + currentIndex;
    sheet.getRange(targetRow, 12).setValue(item.tglAsli);
    sheet.getRange(targetRow, 13).setValue(item.request);
    currentIndex++;
  });
  
  // Tulis data setengah matang di bawahnya agar aman menunggu input kelanjutannya
  listSetengahMatang.forEach((item) => {
    let targetRow = startRow + currentIndex;
    if (item.tglAsli) sheet.getRange(targetRow, 12).setValue(item.tglAsli);
    if (item.request) sheet.getRange(targetRow, 13).setValue(item.request);
    currentIndex++;
  });
  //SpreadsheetApp.getActiveSpreadsheet().toast("📋 Request Konten Update✅");
}

/**
 * TRIGGER WAKTU: Peringatan H-1 dan Hari-H
 * Pasang di menu Trigger (ikon jam) -> cekDeadlineHarian -> Time-driven (Setiap Pagi)
 */
function cekDeadlineHarian() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(CONFIG.SHEET_BRIEF);
  const dbSheet = ss.getSheetByName(CONFIG.SHEET_DATABASE);
  if (!sheet || !dbSheet) return;

  const lastRow = sheet.getLastRow();
  if (lastRow < 4) return;

  const data = sheet.getRange(4, 1, lastRow - 3, 8).getValues();
  const hariIni = new Date();
  hariIni.setHours(0, 0, 0, 0);

  data.forEach((row) => {
    const judul = row[CONFIG.KOLOM.JUDUL - 1];
    const namaPIC = row[CONFIG.KOLOM.PIC - 1];
    const tglTenggat = row[CONFIG.KOLOM.TENGGAT - 1];
    const status = row[CONFIG.KOLOM.PENGERJAAN - 1];

    if (judul && status === "On Progress" && tglTenggat instanceof Date) {
      const tglDeadline = new Date(tglTenggat);
      tglDeadline.setHours(0, 0, 0, 0);
      const selisihHari = Math.ceil((tglDeadline.getTime() - hariIni.getTime()) / (1000 * 3600 * 24));
      
      // REMINDER H-1 (Ke PIC)
      if (selisihHari === 1) {
        kirimKePIC(namaPIC, `⏰ <b>REMINDER H-1</b>\nHalo ${namaPIC}, konten <b>${judul}</b> jatuh tempo besok!`, dbSheet);
      } 
      // TEGURAN HARI-H (Ke PIC & Kepala)
      else if (selisihHari === 0) {
        kirimKePIC(namaPIC, `🚨 <b>DEADLINE HARI INI</b>\nHalo ${namaPIC}, konten <b>${judul}</b> harus selesai hari ini!`, dbSheet);
        kirimKeJabatan(CONFIG.JABATAN_KEPALA, `⚠️ <b>LAPORAN DEADLINE</b>\nKonten <b>${judul}</b> (PIC: ${namaPIC}) belum selesai di hari deadline.`, dbSheet);
      }
    }
  });
}

/**
 * FUNGSI BROADCAST (Kecuali Kepala TIKIM)
 */
function broadcastSemua(pesan, dbSheet) {
  const data = dbSheet.getRange("B2:E" + dbSheet.getLastRow()).getValues();
  data.forEach(row => {
    const jabatan = row[1] ? row[1].toString().trim() : "";
    const idTelegram = row[3];
    if (idTelegram !== "" && jabatan !== CONFIG.JABATAN_KEPALA) {
      executeSend(idTelegram, pesan);
    }
  });
}

  sheet.getRange(4, 1, lastRow - 3, 8).sort({column: 5, ascending: true});

function updatePriorityList() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const kalenderSheet = ss.getSheetByName(CONFIG.SHEET_KALENDER);
  const dashboardSheet = ss.getSheetByName(CONFIG.SHEET_DASHBOARD);
  if (!kalenderSheet || !dashboardSheet) return;
  dashboardSheet.getRange("J3:K7").clearContent();
  const lastRow = kalenderSheet.getLastRow();
  if (lastRow < 5) return;
  const dataKiri = kalenderSheet.getRange(5, 10, lastRow - 4, 2).getValues();
  const styleKiri = kalenderSheet.getRange(5, 11, lastRow - 4, 1).getFontStyles();
  const weightKiri = kalenderSheet.getRange(5, 11, lastRow - 4, 1).getFontWeights();
  const dataKanan = kalenderSheet.getRange(5, 12, lastRow - 4, 2).getValues();
  const styleKanan = kalenderSheet.getRange(5, 13, lastRow - 4, 1).getFontStyles();
  const weightKanan = kalenderSheet.getRange(5, 13, lastRow - 4, 1).getFontWeights();
  let gab = []; const now = new Date(); const cM = now.getMonth(), cY = now.getFullYear();
  const proc = (d, s, w) => {
    for (let i = 0; i < d.length; i++) {
      let t = d[i][0], j = d[i][1];
      if (j && typeof j === "string" && !["Konten", "Request Konten"].includes(j.trim())) {
        let dObj = new Date(t), done = s[i][0] === "italic" && w[i][0] === "bold";
        if (!isNaN(dObj.getTime()) && !done && dObj.getMonth() === cM && dObj.getFullYear() === cY) gab.push([j, dObj]);
      }
    }
  };
  proc(dataKiri, styleKiri, weightKiri); proc(dataKanan, styleKanan, weightKanan);
  gab.sort((a, b) => a[1] - b[1]); let res = gab.slice(0, 5);
  if (res.length > 0) dashboardSheet.getRange(3, 10, res.length, 2).setValues(res);
}

function executeSend(id, p) {
  if (!id) return;
  UrlFetchApp.fetch(`https://api.telegram.org/bot${CONFIG.TOKEN_TELEGRAM}/sendMessage`, {
    "method": "post", "contentType": "application/json",
    "payload": JSON.stringify({ "chat_id": id, "text": p, "parse_mode": "HTML" }), "muteHttpExceptions": true
  });
}

function kirimKePIC(n, p, db) {
  const d = db.getRange("B2:E50").getValues();
  for (let i = 0; i < d.length; i++) { if (d[i][0].toString().trim() === n.toString().trim()) { executeSend(d[i][3], p); break; } }
}

function kirimKeJabatan(j, p, db) {
  const d = db.getRange("B2:E50").getValues();
  for (let i = 0; i < d.length; i++) { if (d[i][1].toString().trim() === j) executeSend(d[i][3], p); }
}

function sinkronisasiKeKalender(s, ks, r, inp) {
  const dk = ks.getRange("J4:M500").getValues(); inp = inp.toString().trim();
  for (let i = 0; i < dk.length; i++) {
    if (dk[i][1].toString().trim() === inp) { s.getRange(r, 5).setValue(dk[i][0]); ks.getRange(i+4, 11).setFontWeight("bold").setFontStyle("normal"); break; }
    else if (dk[i][3].toString().trim() === inp) { s.getRange(r, 5).setValue(dk[i][2]); ks.getRange(i+4, 13).setFontWeight("bold").setFontStyle("normal"); break; }
  }
}

function resetFormatKalender(ks, t) {
  const dk = ks.getRange("J4:M500").getValues(); t = t.toString().trim();
  for (let i = 0; i < dk.length; i++) {
    if (dk[i][1].toString().trim() === t) { ks.getRange(i+4, 11).setFontWeight("normal").setFontStyle("normal"); break; }
    else if (dk[i][3].toString().trim() === t) { ks.getRange(i+4, 13).setFontWeight("normal").setFontStyle("normal"); break; }
  }
}

function setFormatKalenderACC(ks, j) {
  const dk = ks.getRange("J4:M500").getValues(); j = j.toString().trim();
  for (let i = 0; i < dk.length; i++) {
    if (dk[i][1].toString().trim() === j) { ks.getRange(i+4, 11).setFontWeight("bold").setFontStyle("italic"); break; }
    else if (dk[i][3].toString().trim() === j) { ks.getRange(i+4, 13).setFontWeight("bold").setFontStyle("italic"); break; }
  }
}
